 cd папка - куда идём / откуда берём
 git init - стартуем
 git add . - забрать все данные \ git add название_файла - забрать конкретный файл
 git commit -m "комментарий"
 git push origin  - закинуть в гитхаб
 git pull origin  - обновление\забрать документ с гитхаба
 git branch - весь список веток и наша локация в этом списке
 git branch tratata - создание новой ветки "tratatа"
 git checkout название_ветки - переключение веток
 git merge - слияние
 git branch -d - удаление ветки
 
 git push --set-upstream origin название_ветки - это мы из vsc сразу пушим в гит хаб нашу ветку

порядок
git clone https:// ..
cd папка в которой работаем
git branch название ветки
git checkout название ветки (переключение)

сначала pull (обновление) потом push (закинуть):
git checkout название_ветки 
git pull origin название_ветки

git checkout название_ветки
